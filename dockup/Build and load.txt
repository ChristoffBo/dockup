Build From Local

docker stop dockup && docker rm dockup
docker build -t dockup:latest .
docker run -d \
  --name dockup \
  -p 5000:5000 \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v /DATA/Compose:/stacks \
  -v dockup_data:/app/data \
  --restart unless-stopped \
  dockup:latest

Build from Gitea

# Stop and remove the existing container
docker stop dockup && docker rm dockup

# Build the image directly from your private Gitea repo
# Format: http://username:password@host/path/to/repo.git#branch:subdirectory
docker build -t dockup:latest \
  http://Cbothma:Lilly%40057%21@gitea.bothmainc.com/cbothma/dockup_local.git#main:dockup

# Run the freshly built container
docker run -d \
  --name dockup \
  -p 5000:5000 \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v /DATA/Compose:/stacks \
  -v /DATA/AppData:/DATA/AppData:ro \
  -v dockup_data:/app/data \
  --restart unless-stopped \
  dockup:latest
